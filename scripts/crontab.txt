# Sentinel — cron sul server Oracle (ubuntu)
# Installare: copiare in /etc/cron.d/sentinel o crontab -e
# Richiede CRON_SECRET nel file env caricato dal cron (es. source ~/app/.env)

PATH=/usr/local/bin:/usr/bin:/bin
